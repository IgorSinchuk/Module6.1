/* position */
CREATE TABLE `restaurant`.`position` (
`ID` INT NOT NULL AUTO_INCREMENT,
`NAME` VARCHAR(45) NULL,
PRIMARY KEY (`ID`));

/* employee */
CREATE TABLE `restaurant`.`employee` (
`ID` INT NOT NULL AUTO_INCREMENT,
`NAME` VARCHAR(45) NOT NULL,
`AGE` INT NULL,
`SALARY` INT NULL,
`POSITION_ID` INT NULL,
PRIMARY KEY (`ID`),
CONSTRAINT `POSITION_FK`  
FOREIGN KEY (`POSITION_ID`)  
REFERENCES `restaurant`.`position` (`ID`)  
ON DELETE NO ACTION  
ON UPDATE NO ACTION);

/* order */
CREATE TABLE `restaurant`.`order` (
`ID` INT NOT NULL AUTO_INCREMENT,
`TABLE_NUMBER` INT NOT NULL,
`DATE` DATETIME NOT NULL,
`WAITER` INT NOT NULL,
PRIMARY KEY (`ID`),
CONSTRAINT `EMPLOYEE_FK`  
FOREIGN KEY (`WAITER`)  
REFERENCES `restaurant`.`employee` (`ID`) 
ON DELETE NO ACTION  
ON UPDATE NO ACTION);

/* dish */
CREATE TABLE `restaurant`.`dish` (
`ID` INT NOT NULL AUTO_INCREMENT,
`NAME` VARCHAR(45) NOT NULL,`DESCRIPTION` VARCHAR(45) NULL,PRIMARY KEY (`ID`));

/* order_content */
CREATE TABLE `restaurant`.`order_content` (
`ID` INT NOT NULL AUTO_INCREMENT,
`ORDER_ID` INT NOT NULL,
`DISH_ID` INT NOT NULL,
`QUANTITY` INT NOT NULL,
PRIMARY KEY (`ID`),
CONSTRAINT `ORDER_FK`  
FOREIGN KEY (`ORDER_ID`)  
REFERENCES `restaurant`.`order` (`ID`)  
ON DELETE NO ACTION  
ON UPDATE NO ACTION,
CONSTRAINT `DISH_FK`  
FOREIGN KEY (`DISH_ID`)  
REFERENCES `restaurant`.`dish` (`ID`)  
ON DELETE NO ACTION  
ON UPDATE NO ACTION);

/* menu */
CREATE TABLE `restaurant`.`menu` (
`ID` INT NOT NULL AUTO_INCREMENT,
`NAME` VARCHAR(45) NOT NULL,
PRIMARY KEY (`ID`));

/* menu_content */
CREATE TABLE `restaurant`.`menu_content` (`
MENU_ID` INT NOT NULL,
`DISH_ID` INT NOT NULL,
PRIMARY KEY (`MENU_ID`, `DISH_ID`))

/* ingredient */
CREATE TABLE `restaurant`.`ingredient` (
`ID` INT NOT NULL AUTO_INCREMENT,
`NAME` VARCHAR(45) NULL,
PRIMARY KEY (`ID`));

/* dish_components */
CREATE TABLE `restaurant`.`dish_components` (
`DISH_ID` INT NOT NULL,
`INGERDIENT_ID` INT NOT NULL,
`QUANTITY` INT NULL,
PRIMARY KEY (`DISH_ID`, `INGERDIENT_ID`));

/* warehouse */
CREATE TABLE `restaurant`.`warehouse` (
`ID` INT NOT NULL AUTO_INCREMENT,
`INGREDIENT_ID` INT NOT NULL,
`QUANTITY` INT NOT NULL,
PRIMARY KEY (`ID`),
CONSTRAINT `INGREDIENT_FK`  
FOREIGN KEY (`INGREDIENT_ID`)  
REFERENCES `restaurant`.`ingredient` (`ID`)  
ON DELETE NO ACTION  ON UPDATE NO ACTION);

/* kitchen_journal */
CREATE TABLE `restaurant`.`kitchen_journal` (
`ID` INT NOT NULL AUTO_INCREMENT,
`DISH_ID` INT NOT NULL,
 `COOK_ID` INT NOT NULL,
`ORDER_ID` INT NULL,
`DATE` DATETIME NULL,
PRIMARY KEY (`ID`),
CONSTRAINT `DISH`  
FOREIGN KEY (`DISH_ID`)  
REFERENCES `restaurant`.`dish` (`ID`)  
ON DELETE NO ACTION  
ON UPDATE NO ACTION,
CONSTRAINT `COOK`  
FOREIGN KEY (`COOK_ID`)  
REFERENCES `restaurant`.`employee` (`ID`)  
ON DELETE NO ACTION  
ON UPDATE NO ACTION,
CONSTRAINT `ORDER`  
FOREIGN KEY (`ORDER_ID`)  
REFERENCES `restaurant`.`order` (`ID`)  
ON DELETE NO ACTION  
ON UPDATE NO ACTION);
